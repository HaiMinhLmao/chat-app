function connect() { const socket = new SockJS('/ws'); stompClient = Stomp.over(socket); stompClient.connect({}, onConnected, onError); } function onConnected() { console.log('✅ WebSocket Connected'); messageInput.disabled = false; sendBtn.disabled = false; // Subscribe to messages stompClient.subscribe('/topic/messages', onMessageReceived); // Notify others const joinMessage = { sender: userName, content: userName + ' đã tham gia phòng chat!', timestamp: new Date().toISOString() }; stompClient.send('/app/chat.join', {}, JSON.stringify(joinMessage)); } function onError(error) { console.error('❌ WebSocket Error:', error); messageInput.disabled = true; sendBtn.disabled = true; addSystemMessage('Lỗi kết nối. Thử lại...', true); setTimeout(connect, 5000); } function onMessageReceived(payload) { const message = JSON.parse(payload.body); if (message.content.includes('đã tham gia') || message.content.includes('đã rời')) { addSystemMessage(message.content, false); } else if (message.sender === userName) { addMessage(message.content, true, message.sender); } else { addMessage(message.content, false, message.sender); } } function addMessage(content, isSent, sender) { const messageDiv = document.createElement('div'); messageDiv.className = `message ${isSent ? 'sent' : 'received'}`; if (!isSent) { const avatar = document.createElement('div'); avatar.className = 'message-avatar'; avatar.textContent = sender.substring(0, 1).toUpperCase(); messageDiv.appendChild(avatar); } const messageContent = document.createElement('div'); messageContent.className = 'message-content'; messageContent.textContent = content; messageDiv.appendChild(messageContent); const timeDiv = document.createElement('div'); timeDiv.className = 'message-time'; const now = new Date(); timeDiv.textContent = now.getHours().toString().padStart(2, '0') + ':' + now.getMinutes().toString().padStart(2, '0'); messageDiv.appendChild(timeDiv); messagesContainer.appendChild(messageDiv); messagesContainer.scrollTop = messagesContainer.scrollHeight; } function addSystemMessage(content, isError = false) { const messageDiv = document.createElement('div'); messageDiv.className = 'message system'; const contentDiv = document.createElement('div'); contentDiv.className = 'message-content'; contentDiv.style.background = isError ? '#f44336' : '#ffc107'; contentDiv.style.color = isError ? 'white' : '#333'; contentDiv.textContent = content; messageDiv.appendChild(contentDiv); messagesContainer.appendChild(messageDiv); messagesContainer.scrollTop = messagesContainer.scrollHeight; } function sendMessage(event) { if (event) event.preventDefault(); const content = messageInput.value.trim(); console.log('Sending message:', content); if (content === '' || !stompClient || !stompClient.connected) { console.log('Cannot send: empty content or not connected'); return; } const message = { sender: userName, content: content, room: currentRoom, timestamp: new Date().toISOString() }; stompClient.send('/app/chat.send', {}, JSON.stringify(message)); addMessage(content, true, userName); messageInput.value = ''; messageInput.focus(); } // Form submit messageForm.addEventListener('submit', sendMessage); // Conversation item click document.querySelectorAll('.conversation-item').forEach(item => { item.addEventListener('click', function() { document.querySelectorAll('.conversation-item').forEach(i => i.classList.remove('active')); this.classList.add('active'); const conversationName = this.querySelector('.conversation-name').textContent; chatName.textContent = conversationName; messagesContainer.innerHTML = ''; currentRoom = conversationName.toLowerCase(); }); }); // Connect on page load window.addEventListener('load', function() { connect(); addSystemMessage('Phòng chat sẵn sàng. Bắt đầu nhắn tin!'); }); // Cleanup on page close window.addEventListener('beforeunload', function() { if (stompClient && stompClient.connected) { const leaveMessage = { sender: userName, content: userName + ' đã rời khỏi!', timestamp: new Date().toISOString() }; stompClient.send('/app/chat.leave', {}, JSON.stringify(leaveMessage)); stompClient.disconnect(); } });
